⚙️ AWS Client Configuration Bean

@Configuration
public class AwsConfig {

    @Value("${aws.region:us-east-1}")
    private String region;

    /**
     * Credential Provider Chain (in order):
     * 1. Environment variables: AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY
     * 2. Java system properties: aws.accessKeyId, aws.secretAccessKey
     * 3. ~/.aws/credentials file (local dev)
     * 4. EC2 Instance Profile / ECS Task Role / Lambda execution role
     *
     * NEVER hardcode credentials — the chain handles everything automatically.
     */
    @Bean
    public S3Client s3Client() {
        return S3Client.builder()
            .region(Region.of(region))
            .credentialsProvider(DefaultCredentialsProvider.create()) // uses chain
            .build();
    }

    @Bean
    public DynamoDbClient dynamoDbClient() {
        return DynamoDbClient.builder()
            .region(Region.of(region))
            .credentialsProvider(DefaultCredentialsProvider.create())
            .build();
    }

    @Bean
    public SqsClient sqsClient() {
        return SqsClient.builder()
            .region(Region.of(region))
            .credentialsProvider(DefaultCredentialsProvider.create())
            .build();
    }
}

🔐 STS AssumeRole — Cross-Account Access

@Service
public class CrossAccountService {

    private final StsClient stsClient;

    public CrossAccountService() {
        this.stsClient = StsClient.builder()
            .region(Region.US_EAST_1)
            .build();
    }

    /**
     * Assume a role in another AWS account.
     * Use case: your app in Account A needs to access S3 in Account B.
     * NEVER share credentials — use AssumeRole instead.
     */
    public AwsCredentialsProvider assumeRole(String roleArn, String sessionName) {
        AssumeRoleRequest request = AssumeRoleRequest.builder()
            .roleArn(roleArn)                    // "arn:aws:iam::OTHER_ACCOUNT:role/MyRole"
            .roleSessionName(sessionName)         // "my-app-session"
            .durationSeconds(3600)              // 1 hour max by default
            .build();

        AssumeRoleResponse response = stsClient.assumeRole(request);
        Credentials creds = response.credentials();

        return StaticCredentialsProvider.create(
            AwsSessionCredentials.create(
                creds.accessKeyId(),
                creds.secretAccessKey(),
                creds.sessionToken()             // session token required for temp creds
            )
        );
    }

    // Use the assumed role credentials to access another account's S3
    public List<String> listBucketsInOtherAccount(String roleArn) {
        AwsCredentialsProvider crossAccountCreds = assumeRole(roleArn, "list-buckets");

        S3Client crossAccountS3 = S3Client.builder()
            .region(Region.US_EAST_1)
            .credentialsProvider(crossAccountCreds)
            .build();

        return crossAccountS3.listBuckets().buckets().stream()
            .map(Bucket::name)
            .collect(Collectors.toList());
    }
}

☕ Lambda Handler — Best Practices

/**
 * Java Lambda Handler — Production Pattern
 *
 * KEY RULE: Initialize expensive resources OUTSIDE the handler method.
 * The execution context is reused across invocations (warm starts).
 * Static fields and constructor code run only on cold start.
 */
public class OrderHandler implements RequestHandler<APIGatewayProxyRequestEvent, APIGatewayProxyResponseEvent> {

    // ✅ GOOD: initialized once on cold start, reused on warm invocations
    private static final DynamoDbClient dynamoDb;
    private static final ObjectMapper objectMapper;
    private static final String TABLE_NAME;

    static {
        // Static initializer = runs once per container (cold start only)
        TABLE_NAME = System.getenv("TABLE_NAME");  // from Lambda env vars
        dynamoDb = DynamoDbClient.builder()
            .region(Region.of(System.getenv("AWS_REGION")))
            .build();
        objectMapper = new ObjectMapper();
    }

    @Override
    public APIGatewayProxyResponseEvent handleRequest(
            APIGatewayProxyRequestEvent event,
            Context context) {

        // context.getLogger() writes to CloudWatch Logs
        context.getLogger().log("RequestId: " + context.getAwsRequestId());
        context.getLogger().log("Remaining time: " + context.getRemainingTimeInMillis() + "ms");

        try {
            OrderRequest order = objectMapper.readValue(event.getBody(), OrderRequest.class);

            // Process order using pre-initialized DynamoDB client
            Map<String, AttributeValue> item = Map.of(
                "orderId", AttributeValue.fromS(context.getAwsRequestId()),
                "customerId", AttributeValue.fromS(order.getCustomerId()),
                "amount", AttributeValue.fromN(order.getAmount().toString()),
                "status", AttributeValue.fromS("PENDING")
            );

            dynamoDb.putItem(r -> r.tableName(TABLE_NAME).item(item));

            return response(201, "{\"orderId\":\"" + context.getAwsRequestId() + "\"}");

        } catch (Exception e) {
            context.getLogger().log("ERROR: " + e.getMessage());
            return response(500, "{\"error\":\"Internal server error\"}");
        }
    }

    private APIGatewayProxyResponseEvent response(int statusCode, String body) {
        return new APIGatewayProxyResponseEvent()
            .withStatusCode(statusCode)
            .withHeaders(Map.of("Content-Type", "application/json",
                                  "Access-Control-Allow-Origin", "*"))
            .withBody(body);
    }
}

☕ Java Cold Start Tips

  • Java has longer cold starts than Python/Node — use Provisioned Concurrency for latency-sensitive functions
  • Initialize SDK clients in static blocks — reused across warm invocations
  • Use GraalVM Native Image (via Spring Native) for near-zero cold starts
  • Increase memory → more vCPU → faster initialization
  • At 1792MB Lambda gets 1 full vCPU; above that = multiple vCPUs (use multithreading)

🎓 Exam Insights

  • Lambda timeout: default 3s, max 15 minutes (900s)
  • Memory: 128MB–10GB; increasing RAM also increases CPU
  • Initialize DB connections outside handler — reused on warm starts
  • Env variables: 4KB limit; encrypt with KMS for secrets
  • Deployment package: 50MB compressed, 250MB uncompressed (use S3 for larger)

/**
 * StreamLambdaHandler — bridges API Gateway events to Spring Boot.
 * Spring context is initialized ONCE on cold start (static block).
 */
public class StreamLambdaHandler implements RequestStreamHandler {

    private static final SpringBootLambdaContainerHandler<AwsProxyRequest, AwsProxyResponse> handler;

    static {
        try {
            // Cold start: initialize Spring context once
            handler = SpringBootLambdaContainerHandler
                .getAwsProxyHandler(Application.class);
        } catch (ContainerInitializationException e) {
            throw new RuntimeException("Failed to initialize Spring Boot", e);
        }
    }

    @Override
    public void handleRequest(InputStream inputStream,
                               OutputStream outputStream,
                               Context context) throws IOException {
        handler.proxyStream(inputStream, outputStream, context);
    }
}

☕ Lambda Versions & Aliases in Java

// Publish a new version and update alias via SDK
LambdaClient lambda = LambdaClient.builder().region(Region.US_EAST_1).build();

// Publish version from $LATEST
PublishVersionResponse version = lambda.publishVersion(r -> r
    .functionName("order-handler")
    .description("v2 - added validation"));

String newVersion = version.version(); // e.g., "3"

// Canary: 10% to new version, 90% to current prod
lambda.updateAlias(r -> r
    .functionName("order-handler")
    .name("prod")
    .functionVersion("2")                    // current prod version
    .routingConfig(rc -> rc
        .additionalVersionWeights(Map.of(newVersion, 0.1)))); // 10% to v3

🎓 Exam Insights

  • Versions are immutable after publishing — cannot change code or env vars
  • Aliases are mutable pointers — can update which version they point to
  • Aliases cannot reference other aliases
  • Weighted routing on alias = canary deployment
  • API Gateway stage variable: ${stageVariables.lambdaAlias} routes to correct alias
Lab 5 Lambda + SQS — Java Event Source Mapping & Idempotency

▼

Process SQS messages in Java Lambda with proper error handling, idempotency, and batch item failure reporting.

☕ SQS Lambda Handler (Java)

public class SqsOrderProcessor implements RequestHandler<SQSEvent, SQSBatchResponse> {

    private static final DynamoDbClient dynamoDb = DynamoDbClient.create();
    private static final ObjectMapper mapper = new ObjectMapper();

    @Override
    public SQSBatchResponse handleRequest(SQSEvent event, Context context) {
        List<SQSBatchResponse.BatchItemFailure> failures = new ArrayList<>();

        for (SQSEvent.SQSMessage message : event.getRecords()) {
            try {
                processMessage(message, context);
                // Lambda auto-deletes successfully processed messages from SQS
            } catch (Exception e) {
                context.getLogger().log("Failed to process: " + message.getMessageId() + " - " + e.getMessage());
                // Report this specific message as failed — others in batch still succeed
                failures.add(SQSBatchResponse.BatchItemFailure.builder()
                    .withItemIdentifier(message.getMessageId())
                    .build());
            }
        }

        // Return failed items — they go back to queue (or DLQ after MaxReceives)
        return SQSBatchResponse.builder()
            .withBatchItemFailures(failures)
            .build();
    }

    private void processMessage(SQSEvent.SQSMessage message, Context context) throws Exception {
        OrderEvent order = mapper.readValue(message.getBody(), OrderEvent.class);

        // Idempotency check — use messageId as deduplication key
        if (isAlreadyProcessed(message.getMessageId())) {
            context.getLogger().log("Skipping duplicate: " + message.getMessageId());
            return;
        }

        // Process the order
        saveOrder(order);
        markAsProcessed(message.getMessageId());
    }

    private boolean isAlreadyProcessed(String messageId) {
        try {
            GetItemResponse result = dynamoDb.getItem(r -> r
                .tableName("ProcessedMessages")
                .key(Map.of("messageId", AttributeValue.fromS(messageId))));
            return result.hasItem();
        } catch (Exception e) {
            return false;
        }
    }
}
SQS DLQ goes on the SQS queue, NOT on the Lambda function. Lambda DLQ is only for async invocations (S3, SNS triggers).
Spring Cloud AWS alternative: use @SqsListener annotation for cleaner SQS consumption in Spring Boot apps running on EC2/ECS.

🎓 Exam Insights

  • Visibility timeout = 6× Lambda timeout (e.g., Lambda 30s → SQS 180s)
  • SQSBatchResponse with BatchItemFailures = partial batch failure reporting
  • Lambda function should be idempotent — retries must produce same result
  • Lambda scales: adds 60 instances/min for standard SQS, up to 1000 concurrent

☕ Repository Service

@Service
public class OrderRepository {

    private final DynamoDbTable<Order> table;

    public OrderRepository(DynamoDbClient dynamoDbClient) {
        DynamoDbEnhancedClient enhancedClient = DynamoDbEnhancedClient.builder()
            .dynamoDbClient(dynamoDbClient)
            .build();
        this.table = enhancedClient.table("Orders", TableSchema.fromBean(Order.class));
    }

    // CREATE / UPDATE
    public void save(Order order) {
        // Set TTL: expire in 24 hours
        order.setExpiresAt(Instant.now().plusSeconds(86400).getEpochSecond());
        table.putItem(order);
    }

    // READ by primary key
    public Optional<Order> findById(String orderId, String customerId) {
        Key key = Key.builder()
            .partitionValue(orderId)
            .sortValue(customerId)
            .build();
        return Optional.ofNullable(table.getItem(key));
    }

    // CONDITIONAL UPDATE — optimistic locking
    public void updateStatus(String orderId, String customerId,
                              String expectedStatus, String newStatus) {
        Expression condition = Expression.builder()
            .expression("#s = :expected")
            .expressionNames(Map.of("#s", "status"))
            .expressionValues(Map.of(":expected", AttributeValue.fromS(expectedStatus)))
            .build();

        Order order = new Order();
        order.setOrderId(orderId);
        order.setCustomerId(customerId);
        order.setStatus(newStatus);

        table.updateItem(r -> r.item(order).conditionExpression(condition));
        // Throws ConditionalCheckFailedException if status != expectedStatus
    }

    // QUERY with pagination
    public List<Order> findByCustomer(String customerId, int pageSize) {
        QueryConditional queryConditional = QueryConditional
            .keyEqualTo(k -> k.partitionValue(customerId));

        List<Order> results = new ArrayList<>();
        table.query(r -> r
                .queryConditional(queryConditional)
                .limit(pageSize))
            .stream()
            .flatMap(page -> page.items().stream())
            .forEach(results::add);

        return results;
    }

    // BATCH WRITE — up to 25 items
    public void batchSave(List<Order> orders) {
        WriteBatch.Builder<Order> batchBuilder = WriteBatch.builder(Order.class)
            .mappedTableResource(table);
        orders.forEach(batchBuilder::addPutItem);

        DynamoDbEnhancedClient enhancedClient = DynamoDbEnhancedClient.create();
        enhancedClient.batchWriteItem(r -> r.writeBatches(batchBuilder.build()));
    }
}

🎓 Exam Insights

  • WCU: 1 write/s per 1KB. RCU: 1 strong read/s per 4KB (eventual = 2 reads/s)
  • GSI throttle = main table throttle — provision GSI WCU carefully
  • Conditional writes = optimistic locking — throws ConditionalCheckFailedException
  • TTL: set epoch timestamp in Number attribute — deletion within 48 hours
  • Transactions: 2× WCU/RCU cost

☕ DynamoDB Streams Lambda Handler

public class OrderStreamProcessor
        implements RequestHandler<DynamodbEvent, Void> {

    private static final SnsClient sns = SnsClient.create();
    private static final String TOPIC_ARN = System.getenv("ORDER_EVENTS_TOPIC");

    @Override
    public Void handleRequest(DynamodbEvent event, Context context) {

        for (DynamodbEvent.DynamodbStreamRecord record : event.getRecords()) {
            String eventName = record.getEventName(); // INSERT, MODIFY, REMOVE

            switch (eventName) {
                case "INSERT":
                    handleNewOrder(record.getDynamodb().getNewImage(), context);
                    break;
                case "MODIFY":
                    handleOrderUpdate(
                        record.getDynamodb().getOldImage(),
                        record.getDynamodb().getNewImage(),
                        context);
                    break;
                case "REMOVE":
                    handleOrderDeleted(record.getDynamodb().getOldImage(), context);
                    break;
            }
        }
        return null;
    }

    private void handleNewOrder(Map<String, AttributeValue> newImage, Context ctx) {
        String orderId = newImage.get("orderId").getS();
        String customerId = newImage.get("customerId").getS();

        ctx.getLogger().log("New order: " + orderId + " for customer: " + customerId);

        // Publish to SNS for fan-out processing
        sns.publish(r -> r
            .topicArn(TOPIC_ARN)
            .message("{\"event\":\"ORDER_CREATED\",\"orderId\":\"" + orderId + "\"}")
            .subject("New Order"));
    }

    private void handleOrderUpdate(
            Map<String, AttributeValue> oldImage,
            Map<String, AttributeValue> newImage,
            Context ctx) {

        String oldStatus = oldImage.get("status").getS();
        String newStatus = newImage.get("status").getS();

        if (!oldStatus.equals(newStatus)) {
            ctx.getLogger().log("Status changed: " + oldStatus + " → " + newStatus);
            // Trigger notification, update search index, etc.
        }
    }
}
Stream types: KEYS_ONLY, NEW_IMAGE, OLD_IMAGE, NEW_AND_OLD_IMAGES. Choose based on what your processor needs — NEW_AND_OLD_IMAGES for change detection.

🎓 Exam Insights

  • DynamoDB Streams retention: 24 hours fixed (cannot change)
  • Lambda reads streams synchronously via event source mapping
  • Records not retroactively added after enabling streams
  • One Lambda invocation per shard (up to 10 parallel batches per shard)

☕ Secrets Manager — Load DB Credentials at Startup

@Configuration
public class DatabaseConfig {

    private static final SecretsManagerClient secretsClient =
        SecretsManagerClient.create();

    /**
     * Load DB credentials from Secrets Manager.
     * Cache the result — don't call Secrets Manager on every request!
     */
    private static DbCredentials cachedCredentials;

    private static DbCredentials getCredentials() {
        if (cachedCredentials == null) {
            GetSecretValueResponse response = secretsClient.getSecretValue(r -> r
                .secretId("prod/myapp/db"));

            cachedCredentials = new ObjectMapper()
                .readValue(response.secretString(), DbCredentials.class);
        }
        return cachedCredentials;
    }

    @Bean
    @Primary
    public DataSource primaryDataSource() {
        DbCredentials creds = getCredentials();
        HikariConfig config = new HikariConfig();
        config.setJdbcUrl("jdbc:mysql://" + creds.getHost() + ":3306/" + creds.getDbname());
        config.setUsername(creds.getUsername());
        config.setPassword(creds.getPassword());
        config.setMaximumPoolSize(10);          // HikariCP connection pool
        config.setMinimumIdle(2);
        config.setConnectionTimeout(30000);
        config.setIdleTimeout(600000);
        return new HikariDataSource(config);
    }

    // Read Replica DataSource for analytics queries
    @Bean
    public DataSource readReplicaDataSource() {
        DbCredentials creds = getCredentials();
        HikariConfig config = new HikariConfig();
        // Read replica has a different endpoint!
        config.setJdbcUrl("jdbc:mysql://" + creds.getReplicaHost() + ":3306/" + creds.getDbname());
        config.setUsername(creds.getUsername());
        config.setPassword(creds.getPassword());
        config.setMaximumPoolSize(5);
        config.setReadOnly(true);
        return new HikariDataSource(config);
    }
}

☕ Repository with Read/Write Routing

☕ Lazy Loading with @Cacheable (Spring Cache)

@Service
public class ProductService {

    private final ProductRepository productRepository;

    /**
     * Lazy Loading pattern via Spring @Cacheable:
     * 1. Check Redis cache for key "products::productId"
     * 2. Cache HIT → return cached value (fast, no DB call)
     * 3. Cache MISS → call DB, store result in Redis with TTL, return
     */
    @Cacheable(value = "products", key = "#productId")
    public Product getProduct(String productId) {
        // Only called on cache MISS
        return productRepository.findById(productId)
            .orElseThrow(() -> new ProductNotFoundException(productId));
    }

    /**
     * Write-Through pattern: update DB AND evict/update cache
     */
    @CachePut(value = "products", key = "#product.id")
    public Product updateProduct(Product product) {
        return productRepository.save(product); // also updates cache
    }

    // Evict from cache when deleted
    @CacheEvict(value = "products", key = "#productId")
    public void deleteProduct(String productId) {
        productRepository.deleteById(productId);
    }
}

        String message = objectMapper.writeValueAsString(event);

        // Add message attributes for SNS filter policies

            "eventType", MessageAttributeValue.builder()
                .dataType("String")
                .stringValue(event.getEventType())  // e.g., "ORDER_PLACED"

                .stringValue(event.getRegion())

            .topicArn(topicArn)
            .message(message)
            .subject("Order Event: " + event.getEventType())
            .messageAttributes(attributes));

        return response.messageId();

    }

        snsClient.publish(r -> r
            .targetArn(endpointArn)  // device endpoint, not topic
            .message(message)
            .messageStructure("json"));
    }

☕ Kinesis Producer (SDK v2)

@Service
public class ClickstreamProducer {

    private final KinesisClient kinesisClient;
    private final ObjectMapper objectMapper;

    public void publishEvent(ClickEvent event) throws Exception {
        String json = objectMapper.writeValueAsString(event);

        // IMPORTANT: Use diverse partition keys to avoid hot shards!
        // BAD:  partitionKey = "fixed-key"  → all records to one shard
        // GOOD: partitionKey = userId       → distributed across shards
        PutRecordRequest request = PutRecordRequest.builder()
            .streamName("clickstream")
            .data(SdkBytes.fromUtf8String(json))
            .partitionKey(event.getUserId())  // diverse key = even distribution
            .build();

        kinesisClient.putRecord(request);
    }

    // Batch put for higher throughput
    public void publishBatch(List<ClickEvent> events) throws Exception {
        List<PutRecordsRequestEntry> entries = events.stream()
            .map(e -> {
                try {
                    return PutRecordsRequestEntry.builder()
                        .data(SdkBytes.fromUtf8String(objectMapper.writeValueAsString(e)))
                        .partitionKey(e.getUserId())
                        .build();
                } catch (Exception ex) { throw new RuntimeException(ex); }
            })
            .collect(Collectors.toList());

        PutRecordsResponse response = kinesisClient.putRecords(r -> r
            .streamName("clickstream")
            .records(entries));

        // Handle partial failures
        if (response.failedRecordCount() > 0) {
            // Retry failed records with exponential backoff
            retryFailedRecords(response, entries);
        }
    }
}

🎓 Exam Insights

  • Write: 1MB/s or 1000 records/s per shard. Read: 2MB/s per shard
  • Hot shard = bad partition key → use userId, sessionId, not fixed strings
  • KCL: one KCL instance per shard maximum; checkpoints in DynamoDB
  • ProvisionedThroughputExceededException → exponential backoff + add shards
  • Firehose: near real-time (60s), no replay, to S3/Redshift/ES

☕ S3 Service — Full Operations

@Service
public class S3StorageService {

    private final S3Client s3Client;
    private final S3Presigner presigner;

    @Value("${aws.s3.bucket}")
    private String bucketName;

    public S3StorageService(S3Client s3Client) {
        this.s3Client = s3Client;
        this.presigner = S3Presigner.create();
    }

    // Upload with SSE-KMS encryption
    public void uploadEncrypted(String key, byte[] data) {
        s3Client.putObject(r -> r
            .bucket(bucketName)
            .key(key)
            .serverSideEncryption(ServerSideEncryption.AWS_KMS)
            .ssekmsKeyId("alias/my-key"),
            RequestBody.fromBytes(data));
    }

    // Multipart upload for files > 100MB
    public void multipartUpload(String key, Path filePath) throws IOException {
        CreateMultipartUploadResponse init = s3Client.createMultipartUpload(r -> r
            .bucket(bucketName).key(key));
        String uploadId = init.uploadId();

        List<CompletedPart> parts = new ArrayList<>();
        byte[] fileBytes = Files.readAllBytes(filePath);
        int partSize = 5 * 1024 * 1024; // 5MB minimum part size
        int partNumber = 1;

        for (int offset = 0; offset < fileBytes.length; offset += partSize) {
            int length = Math.min(partSize, fileBytes.length - offset);
            byte[] partData = Arrays.copyOfRange(fileBytes, offset, offset + length);
            final int pn = partNumber++;

            UploadPartResponse partResponse = s3Client.uploadPart(r -> r
                .bucket(bucketName).key(key)
                .uploadId(uploadId).partNumber(pn),
                RequestBody.fromBytes(partData));

            parts.add(CompletedPart.builder()
                .partNumber(pn).eTag(partResponse.eTag()).build());
        }

        s3Client.completeMultipartUpload(r -> r
            .bucket(bucketName).key(key).uploadId(uploadId)
            .multipartUpload(m -> m.parts(parts)));
    }

    // Generate pre-signed URL (valid for 1 hour)
    public URL generatePresignedUrl(String key, int expiryMinutes) {
        GetObjectPresignRequest presignRequest = GetObjectPresignRequest.builder()
            .signatureDuration(Duration.ofMinutes(expiryMinutes))
            .getObjectRequest(r -> r.bucket(bucketName).key(key))
            .build();

        return presigner.presignGetObject(presignRequest).url();
        // Anyone with this URL can download the object for expiryMinutes
    }

    // S3 Object Lambda — transform on read (e.g., redact PII)
    public String getObjectViaLambda(String objectLambdaArn, String key) {
        S3Client s3ObjectLambda = S3Client.builder()
            .endpointOverride(URI.create(objectLambdaArn))
            .build();
        ResponseBytes<GetObjectResponse> obj = s3ObjectLambda.getObjectAsBytes(r -> r
            .bucket(bucketName).key(key));
        return obj.asUtf8String(); // transformed by Lambda on the fly
    }
}

🎓 Exam Insights

  • Multipart upload required for files >5GB; recommended for >100MB
  • SSE-S3 header: AES256; SSE-KMS header: aws:kms
  • Pre-signed URLs: time-limited, no AWS credentials needed by recipient
  • S3 Object Lambda: transform data on-the-fly (redact PII, convert format)
  • Transfer Acceleration: uses CloudFront edge locations; bucket name must be DNS-compliant (no dots)

☕ KMS Envelope Encryption in Java

@Service
public class EncryptionService {

    private final KmsClient kmsClient;
    private final String keyId = "alias/my-app-key";

    /**
     * Envelope Encryption:
     * 1. Ask KMS to generate a data key (plaintext + encrypted versions)
     * 2. Encrypt data locally with plaintext data key (fast, no KMS call)
     * 3. Store encrypted data + encrypted data key (NEVER store plaintext key)
     * 4. To decrypt: call KMS to decrypt the data key, then decrypt data locally
     */
    public EncryptedPayload encrypt(String plaintext) throws Exception {
        // Step 1: Generate data key from KMS
        GenerateDataKeyResponse dataKey = kmsClient.generateDataKey(r -> r
            .keyId(keyId)
            .keySpec(DataKeySpec.AES_256));

        byte[] plaintextKey = dataKey.plaintext().asByteArray();
        byte[] encryptedKey = dataKey.ciphertextBlob().asByteArray();

        // Step 2: Encrypt data locally with AES-256
        byte[] encryptedData = aesEncrypt(plaintext.getBytes(), plaintextKey);

        // Step 3: Clear plaintext key from memory immediately!
        Arrays.fill(plaintextKey, (byte) 0);

        return new EncryptedPayload(encryptedData, encryptedKey);
        // Store both encryptedData and encryptedKey — NEVER the plaintext key
    }

    public String decrypt(EncryptedPayload payload) throws Exception {
        // Step 1: Decrypt the data key using KMS
        DecryptResponse decryptedKey = kmsClient.decrypt(r -> r
            .ciphertextBlob(SdkBytes.fromByteArray(payload.getEncryptedKey())));

        byte[] plaintextKey = decryptedKey.plaintext().asByteArray();

        // Step 2: Decrypt data locally
        byte[] decryptedData = aesDecrypt(payload.getEncryptedData(), plaintextKey);
        Arrays.fill(plaintextKey, (byte) 0); // clear from memory

        return new String(decryptedData);
    }
}

// Secrets Manager — cache credentials, don't call on every request
@Component
public class SecretProvider {

    private final SecretsManagerClient secretsClient = SecretsManagerClient.create();
    private final Map<String, String> cache = new ConcurrentHashMap<>();

    public String getSecret(String secretName) {

            GetSecretValueResponse response = secretsClient.getSecretValue(r -> r
                .secretId(name));

        });
    }
}
